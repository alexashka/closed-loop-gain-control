mRetIfF?	macro	link
	call	link
	xorlw	TRUE
	btfss	STATUS,Z
	retlw	FALSE
endm
mFunction	macro	link
			call	link
			endm
mASK		macro	link
			call	link
			endm
mGET		macro	link
			call	link
			endm
mSET		macro	link
			call	link
			endm
mRUN		macro	link
			call	link
			endm
			
mRETURN		macro
			return
endm
